comandos docker

INICIAR/CONTINUAR CONSTRUCCIÓN DE IMAGEN EN DOCKER 
docker-compose build

CONSTRUIR IMAGEN Y ACTIVAR CONTENEDOR
docker-compose up --build (construye y ejecuta)
docker-compose up (activar contenedor)

REVISAR SI EXISTE DOCKER EN EL EQUIPO
docker --version
docker-compose --version

LOGEARSE POR ORACLE
docker login container-registry.oracle.com

ACCEDER AL CONTENEDOR
docker-compose exec oracle-db bash (oracle-db debe coincidir con el nombre del servicio en docker-compose.yml)
docker exec -it oracle_db_course bash (si no llegase a funcionar el comando de arriba)


ACCEDER A LA BASE DE DATOS (UNA VEZ DENTRO PROBAR CON QUERIES)
sqlplus sys/oracle@//localhost:1521/XE as sysdba (sys = nombre usuario, oracle = contraseña, XE = nombre del servicio)

USUARIO = sys
PASSWORD = oracle (revisar ORACLE_PWD en docker-compose.yml)
SID = XE
