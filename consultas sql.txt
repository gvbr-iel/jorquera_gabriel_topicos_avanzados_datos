CREACIÓN DE TABLAS

CREATE TABLE Clientes (
	ClienteID NUMBER PRIMARY KEY,
	Nombre VARCHAR2(50),
	Ciudad VARCHAR2(50),
	FechaNacimiento DATE
	);

CREATE TABLE Pedidos(
	PedidoID NUMBER PRIMARY KEY,
	ClienteID NUMBER,
	Total NUMBER,
	FechaPedido DATE,
	CONSTRAINT fk_pedido_cliente FOREIGN KEY (ClienteID) REFERENCES Clientes(ClienteID)
	);

CREATE TABLE DetallesPedidos(
	DetalleID NUMBER PRIMARY KEY,
	PedidoID NUMBER,
	PorductoID NUMBER,
	Cantidad NUMBER,
	CONSTRAINT fk_detalles_pedido FOREIGN KEY (PedidoID) REFERENCES Pedidos(PedidoID),
	CONSTRAINT fk_detalles_producto FOREGIN KEY (ProductoID) REFERENCES Productos(ProductosID)
	);

SELECT = Selecciona columnas de una tabla
FROM = Especifica la tabla de origen
WHERE = Filtra filas según una condición
ORDER BY = Ordena los resultados

SQL > SELECT Nombre, Ciudad FROM Clientes WHERE Ciudad = 'La Serena' ORDER BY Nombre;

SQL > SELECT PedidoID, Total FROM Pedidos WHERE Total > 500 ORDER BY Total DESC;

SQL > SELECT COUNT(*) AS TotalClientes, Ciudad FROM Clientes GROUP BY Ciudad HAVING COUNT(*) > 2; 

SUBCONSULTAS SQL
CONSULTA ANIDADA DENTRO DE OTRA CONSULTA
SQL > SELECT Nombre FROM Clientes WHERE ClienteID IN (SELECT ClienteID FROM Pedidos WHERE Total > (SELECT AVG(Total) FROM PEDIDOS));

SQL > SELECT Nombre FROM PRODUCTOS WHERE Precio = (SELECT MAX(Precio) FROM Productos);

FUNCIONES DE AGREGACIÓN
COUNT = Cuenta filas
SUM = Suma valores
AVG = Calcula el promedio
MAX/MIN = Encuentra el valor máximo/mínimo

SQL > SELECT Ciudad, COUNT(*) AS TotalClientes FROM Clientes GROUP BY Ciudad HAVING COUNT(*) > 1;

SQL > SELECT c.Nombre, SUM(p.Total) AS TotalGastado, AVG(p.Total) AS PromedioPorPedido FROM Clientes c
LEFT JOIN Pedidos p ON c.ClienteID = p.ClienteID GROUP BY c.Nombre;

ALIAS
ALIAS = Renombrar columnas o tablas temporalmente para mayor claridad en los resultados o consultas.

SQL > SELECT Nombre AS NombreCliente, Ciudad AS Ubicación FROM Clientes;

SQL > c.Nombre, p.Total FROM Clientes c Join Pedidos p ON c.ClienteID = p.ClienteID WHERE p.Total > 500;

EXPRESIONES REGULARES
DEFINICION = Patrones para buscar y filtrar en cadenas de texto

SQL > SELECT Nombre FROM Clientes WHERE REGEXP_LIKE(Nombre, '^J');

SQL > SELECT Nombre, Ciudad FROM Clientes WHERE REGEXP_LIKE(Ciudad, 'ai');

VISTAS
DEFINICION = Consulta almacenada que se comporta como una tabla virtual

SQL > CREATE VIEW nombre_vista AS consulta;

SQL > CREATE VIEW PedidosCaros AS SELECT c.Nombre, p.Total FROM Clientes c JOIN Pedidos p ON c.ClienteID = p.ClienteID WHERE p.Total > 500;
